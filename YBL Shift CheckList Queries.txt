select
  sname.src as userid,
  sname.feed_sys_idd as feed_sys_id,
  to_char(h.hour_start, 'dd/mm/yyyy') as datte,
  to_char(h.hour_start, 'HH:MI:SS AM') as start_time,
  to_char(h.hour_start + interval '1' hour, 'HH:MI:SS AM') as end_time,
  req.reqcont as faa_count,
  ustatus.ucont as us_count,
  fstatus.fcont as fad_count,
  fcase.fcnt as failed_request,
  symaint.mntcnt as system_maint,
  Tokcase.tkncnt as Invalid_token
  
from src_name sname
cross join hours h
left join updt_status ustatus
  on ustatus.source_system = sname.src
  and ustatus.ufid = sname.feed_sys_idd
  and ustatus.time_key = h.hour_start 
LEFT join fad_status fstatus
  on sname.src = fstatus.requested_by
  and fstatus.ffid = sname.feed_sys_idd
  and fstatus.time_key = h.hour_start
left join Token_case Tokcase
  on sname.src=Tokcase.client_requestid
  and Tokcase.request_type = sname.feed_sys_idd
  and Tokcase.time_key = h.hour_start
left join failure_case fcase
  on sname.src=fcase.client_requestid
  and fcase.request_type = sname.feed_sys_idd
  and fcase.time_key = h.hour_start
left join req_status req
  on req.userid = sname.src
  and req.feed_sys_id = sname.feed_sys_idd
  and req.time_key = h.hour_start
left join sys_maint symaint
  on symaint.userid = sname.src
  and symaint.feed_sys_id = sname.feed_sys_idd
  and symaint.time_key = h.hour_start
order by
  h.hour_start,sname.feed_sys_idd asc,case sname.src when 'FINNONEAPS' THEN 1 When 'SFDC' THEN 2 when 'YESRACE' THEN 3 when 'NAVICLOUAPS' THEN 4 when 'DGOD' THEN 5

WITH hours AS (
  SELECT TO_TIMESTAMP('26-SEP-2025','DD-MON-YYYY') + (LEVEL-1) * INTERVAL '1' HOUR AS hour_start
  FROM dual
  CONNECT BY LEVEL <= 24
),
src_name AS (
  SELECT 'SFDCML' AS SRC FROM dual
),
sys_maint AS (
  SELECT TRUNC(insert_time, 'HH') AS time_key, 
         'SFDCML' AS USERID, 
         COUNT(*) AS mntcnt 
  FROM collateraldedupe.psx_ws_audit 
  WHERE TRUNC(insert_time) = TO_DATE('26-SEP-2025','DD-MON-YYYY')
    AND status LIKE '%System under Maintenance%'
  GROUP BY TRUNC(insert_time, 'HH')
),
failure_case AS (
  SELECT TRUNC(insert_time, 'HH') AS time_key,
         'SFDCML' AS USERID,
         COUNT(*) AS fcnt
  FROM collateraldedupe.psx_ws_audit 
  WHERE TRUNC(insert_time) = TO_DATE('26-SEP-2025','DD-MON-YYYY')
    AND status LIKE '%Request Validations Failed%'
  GROUP BY TRUNC(insert_time, 'HH')
),
req_status AS (
  SELECT
    TRUNC(insert_time, 'HH') AS time_key,
    userid,
    COUNT(*) AS reqcont
  FROM collateraldedupe.psx_request
  WHERE TRUNC(insert_time) = TO_DATE('26-SEP-2025','DD-MON-YYYY')

SELECT
  sname.src AS userid,
  TO_CHAR(h.hour_start, 'dd/mm/yyyy') AS datte,
  TO_CHAR(h.hour_start, 'HH:MI:SS AM') AS start_time,
  TO_CHAR(h.hour_start + INTERVAL '1' HOUR, 'HH:MI:SS AM') AS end_time,
  NVL(req.reqcont, 0) AS fc_count, -- Replaced NULL with 0 here
  fcase.fcnt AS failed_request,    -- Remains NULL if empty
  symaint.mntcnt AS system_maint  -- Remains NULL if empty
FROM src_name sname
CROSS JOIN hours h
LEFT JOIN failure_case fcase
  ON fcase.time_key = h.hour_start
  AND fcase.userid = sname.src
LEFT JOIN sys_maint symaint
  ON symaint.time_key = h.hour_start
  AND symaint.userid = sname.src
LEFT JOIN req_status req
  ON req.time_key = h.hour_start
  AND req.userid = sname.src
ORDER BY h.hour_start;
